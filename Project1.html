Problem : 1.2 Reverse Words


							// Reverse Words
							#include <iostream>
							#include <stdio.h>
							#include <string>
							#include <cstring>
							using namespace std;

							void reverse(char *begin, char *end)
							{
							  char temp;
							  while (begin < end)
							  {
								temp = *begin;
								*begin++ = *end;
								*end-- = temp;
							  }
							}

							void reverseWords(char *s)
							{
							  char *word_begin = s;
							  char *temp = s; /* temp is for word boundry */

							  /*STEP 1 of the above algorithm */
							  while( *temp )
							  {
								temp++;
								if (*temp == '\0')
								{
								  reverse(word_begin, temp-1);
								}
								else if(*temp == ' ')
								{
								  reverse(word_begin, temp-1);
								  word_begin = temp+1;
								}
							  } /* End of while */

							   /*STEP 2 of the above algorithm */
							  reverse(s, temp-1);
							}

							int main()
							{
								int i,n;
								string str;

								for (i=1; i<=3; i++){
									cout << "Input Case #" << i <<": ";
									getline(cin, str);
									char *cstr = new char[str.length() + 1];
									strcpy(cstr, str.c_str());
									reverseWords(cstr);
									cout << "Case #" << i <<": " << cstr << endl << endl;
								}
								return 0;
							}
